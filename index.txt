        const brandText = "À LA CHARLOTTE";
        const brandContainer = document.getElementById('brand-name');
        brandText.split('').forEach((char, index) => {
            const span = document.createElement('span');
            span.textContent = char === ' ' ? '\u00A0' : char;
            span.className = 'char-anim';
            span.style.animationDelay = `${index * 0.1}s`;
            brandContainer.appendChild(span);
        });

        const textTarget = "À La Charlotte, l’Art du Pain et de la Gourmandise à Coquelles";
        let characterIndex = 0;
        const typewriterElement = document.getElementById('typewriter');

        function typeWriterEffect() {
            if (characterIndex < textTarget.length) {
                typewriterElement.innerHTML += textTarget.charAt(characterIndex);
                characterIndex++;
                setTimeout(typeWriterEffect, 35); 
            } else {
                const subtitle = document.getElementById('hero-subtitle');
                const button = document.getElementById('hero-button');
                subtitle.classList.remove('opacity-0');
                setTimeout(() => { button.classList.remove('opacity-0'); }, 200);
            }
        }

        window.addEventListener('DOMContentLoaded', () => {
            const bg = document.getElementById('hero-bg');
            if(bg) setTimeout(() => { bg.style.transform = "scale(1)"; }, 100);
            typeWriterEffect();
        });

        function typeScrollEffect(element) {
            if (element.classList.contains('typed-done')) return;
            element.classList.add('typed-done');
            
            const rawText = element.getAttribute('data-text');
            let idx = 0;
            element.innerHTML = "";
            
            function write() {
                if (idx < rawText.length) {
                    element.innerHTML += rawText.charAt(idx);
                    idx++;
                    setTimeout(write, 25); 
                }
            }
            write();
        }

        const itemsToReveal = document.querySelectorAll('.reveal-scale');
        const observer = new IntersectionObserver((entries) => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    entry.target.classList.add('active');
                    
                    const textsToType = entry.target.querySelectorAll('.type-scroll');
                    textsToType.forEach(txt => {
                        setTimeout(() => { typeScrollEffect(txt); }, 300);
                    });
                    
                    observer.unobserve(entry.target);
                }
            });
        }, { threshold: 0.05, rootMargin: '0px 0px -20px 0px' });
        itemsToReveal.forEach(item => observer.observe(item));

        const reviews = [
            { text: "“Excellente boulangerie, les produits sont délicieux et le pain est très bon.”", author: "— Avis Client Google", stars: "★★★★★" },
            { text: "“Les galettes du jour sont excellentes.”", author: "— Avis Client Google", stars: "★★★★★" }
        ];
        let reviewIdx = 0;
        const box = document.getElementById('carousel-item');
        function displayReview() {
            if(!box) return;
            box.style.opacity = '0';
            setTimeout(() => {
                const r = reviews[reviewIdx];
                box.innerHTML = `
                    <p class="text-primary text-sm mb-2">${r.stars}</p>
                    <blockquote class="font-serif italic text-textMain max-w-2xl mx-auto mb-2">${r.text}</blockquote>
                    <cite class="font-hand text-xs text-orangeAccent not-italic font-semibold">${r.author}</cite>
                `;
                box.style.opacity = '1';
                reviewIdx = (reviewIdx + 1) % reviews.length;
            }, 300);
        }
        displayReview();
        setInterval(displayReview, 4500);

        const menuToggle = document.getElementById('menu-toggle');
        const menuClose = document.getElementById('menu-close');
        const mobileMenu = document.getElementById('mobile-menu');

        function toggleMenu() {
            const isOpen = !mobileMenu.classList.contains('translate-x-full');
            if (isOpen) {
                mobileMenu.classList.add('translate-x-full');
                menuToggle.classList.remove('rotate-active');
            } else {
                mobileMenu.classList.remove('translate-x-full');
                menuToggle.classList.add('rotate-active');
            }
        }

        menuToggle.addEventListener('click', toggleMenu);
        menuClose.addEventListener('click', toggleMenu);

        document.querySelectorAll('#mobile-menu nav a').forEach(link => {
            link.addEventListener('click', toggleMenu);
        });

        document.querySelectorAll('.accordion-item').forEach(item => {
            item.addEventListener('click', () => {
                document.querySelectorAll('.accordion-item').forEach(other => {
                    if (other !== item) other.classList.remove('active');
                });
                item.classList.toggle('active');
            });
        });
